function displayPagination() { var paginationDiv = document.createElement('div'); paginationDiv.id = 'pagination'; var totalPages = Math.ceil(lines.length / imagesPerPage); // Create a previous page arrow if (currentPage > 1) { var prevDiv = document.createElement('div'); prevDiv.textContent = '<'; prevDiv.onclick = function() { displayPage(currentPage - 1); }; paginationDiv.appendChild(prevDiv); } // Display page numbers for (let i = 1; i <= totalPages; i++) { var pageNumberDiv = document.createElement('div'); pageNumberDiv.textContent = i; pageNumberDiv.onclick = function() { displayPage(i); }; paginationDiv.appendChild(pageNumberDiv); } // Create a next page arrow if (currentPage < totalPages) { var nextDiv = document.createElement('div'); nextDiv.textContent = '>'; nextDiv.onclick = function() { displayPage(currentPage + 1); }; paginationDiv.appendChild(nextDiv); } document.body.appendChild(paginationDiv); }
